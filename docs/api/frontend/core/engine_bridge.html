106class EngineBridge(QObject):
107    """Wraps BackendEngine and exposes typed dict methods for frontend consumption.
108
109    Emits ``state_updated`` with a ``BridgeContext`` on each timer tick.
110
111    Usage
112    -----
113    .. code-block:: python
114
115        engine = BackendEngine(...)
116        bridge = EngineBridge(engine)
117        bridge.state_updated.connect(my_handler)
118        bridge.start_polling()
119
120    Callers may also access the individual ``get_*()`` methods directly
121    without starting the timer.
122    """
123
124    state_updated = pyqtSignal(BridgeContext)
125
126    def __init__(self, engine: object = None, parent: QObject | None = None) -> None:
127        super().__init__(parent)
128        self._engine = engine
129        self._timer = QTimer(self)
130        self._timer.timeout.connect(self._tick)
131
132    # ------------------------------------------------------------------
133    # Lifecycle
134    # ------------------------------------------------------------------
135
136    def start_polling(self, interval_ms: int = 1000) -> None:
137        """Start the internal tick timer."""
138        self._timer.start(interval_ms)
139
140    def stop_polling(self) -> None:
141        """Stop the internal tick timer."""
142        self._timer.stop()
143
144    def _tick(self) -> None:
145        """Timer callback: emit ``state_updated`` with a fresh BridgeContext."""
146        data = self.get_all()
147        ctx = BridgeContext(data=data, bridge=self)
148        self.state_updated.emit(ctx)

160        except Exception:
161            return {}

173    def get_cpu_metrics(self) -> CpuMetricsDict:
174        """CPU usage, per-core breakdown, frequency and core counts."""
175        state = self._state
176        cpu = state.get("cpu", {})
177        assert isinstance(cpu, dict)
178        return CpuMetricsDict(
179            cpu_percent=cpu.get("usage_percent", 0.0),  # type: ignore[arg-type]
180            per_core=[],
181            frequency=None,
182            physical_cores=cpu.get("physical_cores", 0),  # type: ignore[arg-type]
183            logical_cores=cpu.get("logical_cores", 0),  # type: ignore[arg-type]
184        )
185
186    def get_memory_metrics(self) -> MemoryMetricsDict:
187        """RAM totals, usage, available, free, cached and percent."""
188        state = self._state
189        ram = state.get("ram", {})
190        assert isinstance(ram, dict)
191        total = ram.get("total_bytes", 0)
192        used = ram.get("used_bytes", 0)
193        available = ram.get("available_bytes", 0)
194        percent = ram.get("percent", 0.0)
195        return MemoryMetricsDict(
196            total=total,  # type: ignore[arg-type]
197            used=used,  # type: ignore[arg-type]
198            available=available,  # type: ignore[arg-type]
199            free=available,  # engine reports 'available' as free
200            cached=0,
201            percent=percent,  # type: ignore[arg-type]
202        )
203
204    def get_disk_usage(self, path: str) -> DiskUsageDict:
205        """Usage stats for *path* (total / used / free bytes + percent)."""
206        state = self._state
207        storage_list = state.get("storage", [])
208        assert isinstance(storage_list, list)
209        for disk in storage_list:
210            if isinstance(disk, dict) and disk.get("mount_point", "") == path:
211                return DiskUsageDict(
212                    total=disk.get("total_bytes", 0),  # type: ignore[arg-type]
213                    used=disk.get("used_bytes", 0),  # type: ignore[arg-type]
214                    free=disk.get("free_bytes", 0),  # type: ignore[arg-type]
215                    percent=disk.get("percent", 0.0),  # type: ignore[arg-type]
216                )
217        return DiskUsageDict(total=0, used=0, free=0, percent=0.0)
218
219    def get_network_io(self) -> NetworkIODict:
220        """Aggregate bytes sent / received across all interfaces."""
221        state = self._state
222        net_list = state.get("network", [])
223        assert isinstance(net_list, list)
224        total_sent = 0
225        total_recv = 0
226        for iface in net_list:
227            if isinstance(iface, dict):
228                total_sent += iface.get("bytes_sent", 0)
229                total_recv += iface.get("bytes_recv", 0)
230        return NetworkIODict(bytes_sent=total_sent, bytes_recv=total_recv)

232    def get_process_list(self) -> list[ProcessEntryDict]:
233        """Snapshot of running processes (limited fields)."""
234        state = self._state
235        proc_list = state.get("processes", [])
236        assert isinstance(proc_list, list)
237        result: list[ProcessEntryDict] = []
238        for proc in proc_list:
239            if isinstance(proc, dict):
240                result.append(
241                    ProcessEntryDict(
242                        pid=proc.get("pid", 0),  # type: ignore[arg-type]
243                        name=proc.get("name", ""),  # type: ignore[arg-type]
244                        cpu_percent=proc.get("cpu_percent", 0.0),  # type: ignore[arg-type]
245                        memory_info=proc.get("memory_rss", 0),  # type: ignore[arg-type]
246                        status=proc.get("status", ""),  # type: ignore[arg-type]
247                        num_threads=proc.get("num_threads", 0),  # type: ignore[arg-type]
248                        username=proc.get("username"),  # type: ignore[arg-type]
249                        ppid=proc.get("ppid"),  # type: ignore[arg-type]
250                        create_time=proc.get("create_time"),  # type: ignore[arg-type]
251                        exe=proc.get("exe"),  # type: ignore[arg-type]
252                    )
253                )
254        return result

256    def get_sensors(self) -> dict[str, list[float]]:
257        """Temperatures keyed by sensor name → list of values."""
258        state = self._state
259        sensor_list = state.get("sensors", [])
260        assert isinstance(sensor_list, list)
261        temps: dict[str, list[float]] = {}
262        for s in sensor_list:
263            if isinstance(s, dict):
264                name = str(s.get("name", "unknown"))
265                value = float(s.get("value", 0.0))
266                temps.setdefault(name, []).append(value)
267        return temps

269    def get_system_load(self) -> SystemLoadDict:
270        """CPU load percent, process / thread / handle counts."""
271        state = self._state
272        cpu = state.get("cpu", {})
273        assert isinstance(cpu, dict)
274        proc_list = state.get("processes", [])
275        assert isinstance(proc_list, list)
276        return SystemLoadDict(
277            cpu_percent=cpu.get("usage_percent", 0.0),  # type: ignore[arg-type]
278            processes=len(proc_list) if isinstance(proc_list, list) else 0,
279            threads=0,
280            handles=0,
281        )

283    def get_static_info(self) -> StaticInfoDict:
284        """Static system info from the active driver (or defaults)."""
285        driver = self._driver
286        static = None
287        if driver is not None and hasattr(driver, "get_static_info"):
288            try:
289                static = driver.get_static_info()  # type: ignore[union-attr, reportAttributeAccessIssue]
290            except Exception:
291                pass
292        if static is not None:
293            if hasattr(static, "model_dump"):
294                raw = static.model_dump()
295            else:
296                raw = dict(static)  # type: ignore[arg-type]
297            return StaticInfoDict(
298                hostname=raw.get("hostname", ""),
299                os_name=raw.get("os_name", ""),
300                os_version=raw.get("os_version", ""),
301                architecture=raw.get("architecture", ""),
302                cpu_brand=raw.get("cpu_brand", ""),
303                cpu_physical_cores=raw.get("cpu_physical_cores", 0),
304                cpu_logical_cores=raw.get("cpu_logical_cores", 0),
305                cpu_frequency_mhz=raw.get("cpu_frequency_mhz"),
306                total_ram_bytes=raw.get("total_ram_bytes", 0),
307                python_version=raw.get("python_version", ""),
308                boot_time=raw.get("boot_time", ""),
309            )
310        return StaticInfoDict(
311            hostname="",
312            os_name="",
313            os_version="",
314            architecture="",
315            cpu_brand="",
316            cpu_physical_cores=0,
317            cpu_logical_cores=0,
318            cpu_frequency_mhz=None,
319            total_ram_bytes=0,
320            python_version="",
321            boot_time="",
322        )

324    def get_boot_time(self) -> float:
325        """Boot timestamp as a float (or 0.0 when unavailable)."""
326        return 0.0

328    def get_disk_partitions(self) -> list[dict[str, str]]:
329        """Partition list derived from engine storage data."""
330        state = self._state
331        storage_list = state.get("storage", [])
332        assert isinstance(storage_list, list)
333        return [
334            {
335                "device": "",
336                "mountpoint": s.get("mount_point", "") if isinstance(s, dict) else "",
337                "fstype": "",

339            for s in (storage_list or [])
340        ]

343        """Network interface → addresses (not yet provided by engine)."""

346    def get_battery(self) -> BatteryDict:
347        """Battery charge / status dict."""
348        state = self._state
349        bat = state.get("battery")
350        if bat is None:
351            return BatteryDict(percent=0.0, power_plugged=None, seconds_left=None)
352        if isinstance(bat, dict):
353            return BatteryDict(
354                percent=bat.get("percent", 0.0),  # type: ignore[arg-type]
355                power_plugged=bat.get("power_plugged"),  # type: ignore[arg-type]
356                seconds_left=bat.get("seconds_left"),  # type: ignore[arg-type]
357            )
358        return BatteryDict(percent=0.0, power_plugged=None, seconds_left=None)

365        """Aggregate all metrics into a single TypedDict."""
366        return AggregatedStateDict(

369            disks=[self.get_disk_usage("/")],

383    def terminate_process(self, pid: int) -> bool:
384        """Request graceful process termination.  Returns success."""
385        return self._manage_process(pid, "terminate")

387    def kill_process(self, pid: int) -> bool:
388        """Force-kill a process.  Returns success."""
389        return self._manage_process(pid, "kill")

391    def _manage_process(self, pid: int, action: str) -> bool:
392        driver = self._driver
393        if driver is not None and hasattr(driver, "manage_process"):
394            try:
395                return bool(driver.manage_process(pid, action))  # type: ignore[union-attr, reportAttributeAccessIssue]
396            except Exception:
397                pass
398        return False
